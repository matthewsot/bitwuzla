macro(add_test_case option testcase)
  string(REPLACE " " "_" TESTNAME ${testcase})
  string(REPLACE "-" "" TESTNAME ${TESTNAME})
  set(tcase "${CMAKE_CURRENT_LIST_DIR}/${testcase}")
  add_test(${TESTNAME}
      ${RUN_TEST_CASE_SCRIPT}
      ${option}
      ${BITWUZLA_BINARY}
      ${tcase}
      ${CMAKE_CURRENT_BINARY_DIR}
  )
  set_tests_properties(${TESTNAME} PROPERTIES LABELS "regress")
endmacro()
